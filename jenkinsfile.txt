pipeline {
    agent any

    environment {
        PROJECT_NAME = "my-python-project"
        PYVER = "3.10"
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/sample.git'
            }
        }

        stage('Build') {
            steps {
                sh """
                    python${env.PYVER} -m venv venv
                    source venv/bin/activate
                    pip install --upgrade pip
                    pip install -r requirements.txt || true
                """
            }
        }

        stage('Test') {
            steps {
                sh """
                    source venv/bin/activate
                    pytest || echo "No tests found"
                """
            }
        }

        stage('Deploy') {
            steps {
                echo "Deploying ${env.PROJECT_NAME}..."
            }
        }
    }
}
